                                      Add Content Item

                                  </div>
                                </div>

                            </div>
                          );

                    )}
                  </div>

              })}
            </div>

    </div>

            className="w-full border rounded px-3 py-2"

            className="px-4 py-2 text-sm bg-gray-100 rounded hover:bg-gray-200"
          >
            Cancel
          </button>
          <button
            type="submit"
            className="px-5 py-2 text-sm bg-blue-600 text-white rounded hover:bg-blue-700"
          >
            Add Content
          </button>
        </div>

    <Modal title="Add Shelf (Grade Band)" onClose={() => setShowAddShelfModal(false)}>
      <form onSubmit={submitAddShelf} className="space-y-4">
        <div>
          <label className="text-xs font-semibold text-gray-600 block mb-1">Shelf Label</label>

            value={shelfForm.shelf}
            onChange={(e) => setShelfForm({ shelf: e.target.value })}
            placeholder="e.g., 9-10 or PreK"
            required
            className="w-full border rounded px-3 py-2"
          />
        </div>
        <div className="flex justify-end space-x-2 pt-2">
          <button
            type="button"
            onClick={() => setShowAddShelfModal(false)}
            className="px-4 py-2 text-sm bg-gray-100 rounded hover:bg-gray-200"
          >
            Cancel
          </button>

</div>
