    <script>
        let activeFilters = ['math', 'art', 'invention', 'myth', 'power', 'sel'];
        
        function toggleFilter(btn) {
            const category = btn.dataset.category;
            btn.classList.toggle('active');
            
            if (activeFilters.includes(category)) {
                activeFilters = activeFilters.filter(f => f !== category);
            } else {
                activeFilters.push(category);
            }
            
            updateDisplay();
        }
        
        function updateDisplay() {
            const gradeFilter = document.getElementById('gradeSelect').value;
            
            // Update timeline eras
            document.querySelectorAll('.era').forEach(era => {
                const grades = era.dataset.grades ? era.dataset.grades.split(',') : ['all'];
                const gradeMatch = gradeFilter === 'all' || grades.includes(gradeFilter);
                
                if (gradeMatch) {
                    era.style.opacity = '1';
                    era.style.pointerEvents = 'auto';
                    
                    // Update categories within era
                    era.querySelectorAll('.category').forEach(cat => {
                        const catType = Array.from(cat.classList).find(c => 
                            ['math', 'art', 'invention', 'myth', 'power', 'sel'].includes(c)
                        );
                        cat.style.display = activeFilters.includes(catType) ? 'block' : 'none';
                    });
                } else {
                    era.style.opacity = '0.3';
                    era.style.pointerEvents = 'none';
                }
            });
            
            // Update lesson cards
            document.querySelectorAll('.lesson-card').forEach(card => {
                const cardGrade = card.dataset.grade;
                if (gradeFilter === 'all' || cardGrade === gradeFilter) {
                    card.style.display = 'block';
                } else {
                    card.style.display = 'none';
                }
            });
        }
        
        // Grade selector event listener
        document.getElementById('gradeSelect').addEventListener('change', updateDisplay);
        
        // Initialize display
        updateDisplay();
        
        // Add smooth scrolling to timeline
        const timeline = document.querySelector('.timeline-container');
        let isDown = false;
        let startX;
        let scrollLeft;
        
        timeline.addEventListener('mousedown', (e) => {
            isDown = true;
            startX = e.pageX - timeline.offsetLeft;
            scrollLeft = timeline.scrollLeft;
            timeline.style.cursor = 'grabbing';
        });
        
        timeline.addEventListener('mouseleave', () => {
            isDown = false;
            timeline.style.cursor = 'grab';
        });
        
        timeline.addEventListener('mouseup', () => {
            isDown = false;
            timeline.style.cursor = 'grab';
        });
        
        timeline.addEventListener('mousemove', (e) => {
            if (!isDown) return;
            e.preventDefault();
            const x = e.pageX - timeline.offsetLeft;
            const walk = (x - startX) * 2;
            timeline.scrollLeft = scrollLeft - walk;
        });
    </script>
